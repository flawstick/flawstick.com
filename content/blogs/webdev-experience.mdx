# The Webdev Experience

As a former low level language and security expert, going into web development felt unnatural. In this post I offer a unique outlook on the transition between kernel level development and security scrutiny to the world of web development.

## What is it?

Henoamly is a multi-modal for anomaly detection in hebrew scriptures. This model employs a multi-modal approach toward anomaly detection, utilizing cutting-edge technologies such as Vision Transformers (ViT) with their attention to diminutive intricacies, as-well as a highly complex LSTM structure for Optical Character Recognition. Furthermore, it is capable of accurately adjusting preprocessing parameters to achieve OCR-eligible images.

## What does it solve?

Hours go into manual anomaly detection in letters and words in hebrew
scriptures, approximately seven minutes per page. This model is capable of reducing
that amount drastically, even on a low-performance core processing unit.

export default function Blog() {
  const [isLoaded, setIsLoaded] = useState(false)
  const [blogs, setBlogs] = useState(blogsData)
  const [filteredBlogs, setFilteredBlogs] = useState(blogsData)
  const [views, setViews] = useState({})
  const [viewsLoaded, setViewsLoaded] = useState(false)

  useEffect(() => {
    // Fetch view counts from Redis
    async function fetchViewCounts() {
      try {
        const slugs = blogs.map((p) => p.slug)
        const viewCounts = await getMultipleBlogViews(slugs)
        setViews(viewCounts)
        setViewsLoaded(true)
      } catch (error) {
        console.error("Error fetching view counts:", error)
        // Fallback to empty view counts
        const slugs = blogs.map((p) => p.slug)
        setViews(
          slugs.reduce((acc, slug) => {
            acc[slug] = 0
            return acc
          }, {}),
        )
        setViewsLoaded(true)
      }
    }

    fetchViewCounts()

    // Filter blogs based on search term
    setFilteredBlogs(blogs)

    // Add a small delay before showing content
    setTimeout(() => {
      setIsLoaded(true)
    }, 300)
  }, [blogs])

  if (!isLoaded) {
    return (
      <div className="relative pb-20 mt-8">
        <div className="px-6 pt-20 mx-auto space-y-8 max-w-7xl lg:px-8 md:space-y-16 md:pt-24 lg:pt-32">
          <div className="max-w-2xl mx-auto lg:mx-0">
            <Skeleton className="h-10 w-48 mb-4" />
            <Skeleton className="h-4 w-full max-w-md" />
          </div>
          <div className="w-full h-px bg-zinc-800" />
          <div className="grid grid-cols-1 gap-8 mx-auto lg:grid-cols-2">
            <Skeleton className="h-[400px] w-full rounded-xl" />
            <div className="flex flex-col w-full gap-8 mx-auto">
              <Skeleton className="h-[180px] w-full rounded-xl" />
              <Skeleton className="h-[180px] w-full rounded-xl" />
            </div>
          </div>
        </div>
      </div>
    )
  }

  const featured = blogs.find((blog) => blog.slug === "neovim-vs-cursor")
  const top2 = blogs.find((blog) => blog.slug === "ai-coding-assistants")
  const top3 = blogs.find((blog) => blog.slug === "minimalist-web-design")
  const sorted = blogs
    .filter((p) => p.published)
    .filter((blog) => blog.slug !== featured.slug && blog.slug !== top2.slug && blog.slug !== top3.slug)
    .sort(
      (a, b) =>
        new Date(b.date ?? Number.POSITIVE_INFINITY).getTime() - new Date(a.date ?? Number.POSITIVE_INFINITY).getTime(),
    )

  return (
    <div className="relative pb-20 mt-8">
      <div className="px-6 pt-20 pb-4 mx-auto space-y-8 max-w-7xl lg:px-8 md:space-y-16 md:pt-24 lg:pt-28">

                        {featured.date ? (

                          <span>SOON</span>
                        )}
                      </div>
                      <span className="flex items-center gap-1 text-xs text-zinc-500">
                        <Eye className="w-4 h-4" />{" "}
                        {viewsLoaded ? (
                          Intl.NumberFormat("en-US", { notation: "compact" }).format(views[featured.slug] ?? 0)
                        ) : (
                          <span className="h-3 w-6 bg-zinc-800 animate-pulse rounded"></span>
                        )}
                      </span>
                    </div>

                    <h2
                      id="featured-post"
                      className="mt-4 text-3xl font-bold text-zinc-100 group-hover:text-white sm:text-4xl font-display"
                    >
                      {featured.title}
                    </h2>
                    <p className="mt-4 leading-8 duration-150 text-zinc-400 group-hover:text-zinc-300">
                      {featured.description}
                    </p>
                    <div className="absolute bottom-4 md:bottom-8">
                      <p className="hidden group text-zinc-200 hover:text-zinc-50 duration-500 lg:block">
                        Read more <span aria-hidden="true">&rarr;</span>

              <div className="grid grid-cols-1 gap-4 mx-auto lg:mx-0 md:grid-cols-3">

                    .filter((_, i) => i % 3 === 0)
                    .map((blog) => (
                      <Card key={blog.slug}>
                        <Link href={`/blog/${blog.slug}`}>
                          <BlogArticle blog={blog} views={viewsLoaded ? (views[blog.slug] ?? 0) : null} />
                        </Link>
                      </Card>
                    ))}
                </div>

                  {sorted
                    .filter((_, i) => i % 3 === 1)
                    .map((blog) => (
                      <Card key={blog.slug}>
                        <Link href={`/blog/${blog.slug}`}>
                          <BlogArticle blog={blog} views={viewsLoaded ? (views[blog.slug] ?? 0) : null} />
                        </Link>
                      </Card>
                    ))}
                </div>
                <div className="grid grid-cols-1 gap-4">
                  {sorted
                    .filter((_, i) => i % 3 === 2)

                      <Card key={blog.slug}>
                        <Link href={`/blog/${blog.slug}`}>
                          <BlogArticle blog={blog} views={viewsLoaded ? (views[blog.slug] ?? 0) : null} />
                        </Link>
                      </Card>
                    ))}
                </div>
              </div>
            )}
          </>
        ) : (
          <div className="text-center py-12 text-gray-400 animate-fadeIn">No posts found matching your search.</div>
        )}
      </div>
    </div>
  )
}
